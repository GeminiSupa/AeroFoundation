## 🎯 How to Use the New Structure

### 1. **Use the API Layer**

Instead of:
```typescript
// Old way - direct mock data
const [students, setStudents] = useState(mockStudents);
```

Do this:
```typescript
// New way - use API layer
import { getStudents } from '@/lib/api/students';

const fetchStudents = async () => {
  const result = await getStudents(page, pageSize);
  if (result.success) {
    setStudents(result.data.data);
  } else {
    toast.error(result.error);
  }
};
```

### 2. **Type Everything**

```typescript
import type { Student, ApiResponse } from '@/types';

const [student, setStudent] = useState<Student | null>(null);
const [loading, setLoading] = useState(false);

const fetchStudent = async (id: string) => {
  setLoading(true);
  const result: ApiResponse<Student> = await getStudentById(id);
  setLoading(false);
  
  if (result.success) {
    setStudent(result.data!);
  }
};
```

### 3. **Handle Errors Consistently**

```typescript
try {
  const result = await addStudent(formData);
  
  if (result.success) {
    toast.success(result.message || 'Student added!');
    refreshStudents();
    closeModal();
  } else {
    toast.error(result.error);
  }
} catch (error) {
  toast.error('Unexpected error occurred');
  console.error(error);
}
```

### 4. **Use AI Insights**

```typescript
import { AIInsights } from '@/components/AIInsights';

// In any dashboard
<AIInsights role="teacher" />  // or "admin", "student", "parent"
```
